        // Initial default parameters
        const DEFAULTS = { L: 420, k: 0.08, x0: 60 };

        // MATH EQUATIONS RENDERING
        function updateMathFormulas(L, k, x0) {
            if (typeof katex === 'undefined') return;
            
            // Format parameters to avoid double signs
            const x0Str = x0 >= 0 ? `-${x0}` : `+${Math.abs(x0)}`;
            
            const fxLatex = `f(x) = \\frac{${L}}{1 + e^{-${k}(x ${x0Str})}}`;
            const dfxLatex = `f'(x) = \\frac{${L} \\cdot ${k} \\cdot e^{-${k}(x ${x0Str})}}{(1 + e^{-${k}(x ${x0Str})})^2}`;
            
            katex.render(fxLatex, document.getElementById('latex-fx'), { throwOnError: false });
            katex.render(dfxLatex, document.getElementById('latex-dfx'), { throwOnError: false });
        }

        // CUMULATIVE VALUE CALCULATION f(x)
        function f(x, L, k, x0) {
            return L / (1 + Math.exp(-k * (x - x0)));
        }

        // DERIVATIVE CALCULATION f'(x)
        function df(x, L, k, x0) {
            const expTerm = Math.exp(-k * (x - x0));
            return (L * k * expTerm) / Math.pow(1 + expTerm, 2);
        }

        // GENERATE DATASETS FOR CHART
        function generateData(L, k, x0) {
            const days = [];
            const weights = [];
            const rates = [];

            for (let x = 0; x <= 120; x++) {
                days.push(x);
                weights.push(parseFloat(f(x, L, k, x0).toFixed(2)));
                rates.push(parseFloat(df(x, L, k, x0).toFixed(2)));
            }

            return { days, weights, rates };
        }

        // ADVISORY LOGIC ACCORDING TO THE DEVELOPMENT STAGE
        function getBioscienceAdvisory(day, weight, rate, inflectionDay) {
            let desc = "";
            let stage = "";
            
            if (day < inflectionDay - 15) {
                stage = "Cell Division & Lag Growth Phase (Fruit Set)";
                desc = `During this early stage (${day} DAA), the mango grows very slowly at only ${rate} g/day. Energy is prioritized for cell division rather than expansion. 
                <strong class="text-emerald-800">SFA Advice:</strong> Focus on balanced Boron, Calcium, and optimal water supply. Foliar fertilizer applications are critical to set strong fruit cellular frameworks. Avoid early nitrogen shock to prevent fruit drop.`;
            } else if (day >= inflectionDay - 15 && day <= inflectionDay + 15) {
                stage = "Exponential Cell Expansion Phase (Peak Growth)";
                desc = `You are inspecting the crucial fast-growth window. Growth rate is high (${rate} g/day), peaking around Day ${inflectionDay}. 
                <strong class="text-emerald-800">SFA Advice:</strong> Potassium (K) demands are at maximum to pull sugars into the growing fruit cells. Drought stress or waterlogging right now causes fruit cracking or massive size depletion. Daily precision micro-irrigation is recommended by DOA Perlis manuals.`;
            } else {
                stage = "Maturation & Ripening Phase (Plateau)";
                desc = `At ${day} DAA, cumulative weight starts to level off at ${weight}g, and growth rate declines significantly to ${rate} g/day. Cells are shifting from water accumulation to organic acid decomposition and starch-to-sugar conversion. 
                <strong class="text-emerald-800">SFA Advice:</strong> Halt nitrogen and reduce water. High nitrogen at this stage causes the "spongy tissue" physiological disorder in Harumanis mangos. Monitor skin color, shoulder slope, and latex flow for harvesting.`;
            }

            return `<strong>Stage: ${stage}</strong><br><span class="mt-1 block">${desc}</span>`;
        }

        // UPDATE THE DATA DISPLAY PANEL (INSPECTOR)
        function updateInspector(day, L, k, x0) {
            const w = f(day, L, k, x0);
            const r = df(day, L, k, x0);

            inspectDayBadge.textContent = `Day ${day}`;
            inspectWeight.textContent = `${w.toFixed(2)}g`;
            inspectRate.textContent = `${r.toFixed(2)}g/day`;
            inspectSlope.textContent = `m = +${r.toFixed(4)}`;
            inspectAdvice.innerHTML = getBioscienceAdvisory(day, w.toFixed(2), r.toFixed(2), x0);

            // Dynamically scale and color the SVG Mango based on development stage
            const maxW = L;
            const percentOfMax = Math.min((w / maxW), 1.0);
            
            // Scale between 0.25 (tiny fruitlet) and 1.1 (mature fruit)
            const scaleFactor = 0.25 + (percentOfMax * 0.85);
            miniMangoSvg.style.transform = `scale(${scaleFactor})`;

            // Color shift: Early lag (deep green), Inflection (light lime green), Ripening (golden mango yellow)
            let color = "#15803d"; // Default deep green
            if (day > 45 && day <= 80) {
                color = "#84cc16"; // Lime green during rapid expansion
            } else if (day > 80) {
                // Blend from lime green to rich mango yellow
                color = "#eab308"; // Golden yellow
            }
            mangoBody.setAttribute("fill", color);
        }

        // UPDATE CHART DATA DYNAMICALLY
        function updateChart(data) {
            if (!myChart) return;
            myChart.data.labels = data.days;
            myChart.data.datasets[0].data = data.weights;
            myChart.data.datasets[1].data = data.rates;
            myChart.update();
        }

        // HIGHLIGHT SELECTED NODE (custom helper using chart.js config)
        function highlightSelectedPoint(dayIndex) {
            if (!myChart) return;
            // Clear any custom styles by modifying normal state arrays
            const styles = myChart.data.datasets.map(d => {
                const radii = new Array(d.data.length).fill(1);
                radii[dayIndex] = 8; // larger radius for selected index
                return radii;
            });
            
            myChart.data.datasets[0].pointRadius = styles[0];
            myChart.data.datasets[1].pointRadius = styles[1];
            myChart.update('none'); // silent update without resetting layout state
        }

        // SLIDER CHANGE HANDLERS
        function handleParamsChange() {
            const L = parseFloat(sliderL.value);
            const k = parseFloat(sliderK.value);
            const x0 = parseFloat(sliderX0.value);

            // Update DOM text
            valL.textContent = `${L}g`;
            valK.textContent = k;
            valX0.textContent = `${x0} DAA`;

            // Regenerate dataset and update chart & formulas
            const data = generateData(L, k, x0);
            updateChart(data);
            updateMathFormulas(L, k, x0);

            // Update active inspector values
            updateInspector(selectedDay, L, k, x0);
            highlightSelectedPoint(selectedDay);
        }

        sliderL.addEventListener('input', handleParamsChange);
        sliderK.addEventListener('input', handleParamsChange);
        sliderX0.addEventListener('input', handleParamsChange);

        // RUN ACTIVE SIMULATION SEQUENCE
        function animateGrowth() {
            const L = parseFloat(sliderL.value);
            const k = parseFloat(sliderK.value);
            const x0 = parseFloat(sliderX0.value);
            
            let day = 0;
            btnAnimate.disabled = true;
            btnAnimate.classList.add('opacity-50', 'cursor-not-allowed');

            const interval = setInterval(() => {
                selectedDay = day;
                updateInspector(day, L, k, x0);
                highlightSelectedPoint(day);
                
                day++;
                if (day > 120) {
                    clearInterval(interval);
                    btnAnimate.disabled = false;
                    btnAnimate.classList.remove('opacity-50', 'cursor-not-allowed');
                }
            }, 30); // 30ms per step = ~3.6s total growth cycle simulation
        }

        btnAnimate.addEventListener('click', animateGrowth);

        // RESET DEFAULT SIMULATION PARAMETERS
        btnReset.addEventListener('click', () => {
            sliderL.value = DEFAULTS.L;
            sliderK.value = DEFAULTS.k;
            sliderX0.value = DEFAULTS.x0;
            selectedDay = 60;
            handleParamsChange();
        });

        // INITIATE APP ON LOAD
        window.onload = function() {
            const initialData = generateData(DEFAULTS.L, DEFAULTS.k, DEFAULTS.x0);
            initChart(initialData);
            updateMathFormulas(DEFAULTS.L, DEFAULTS.k, DEFAULTS.x0);
            updateInspector(selectedDay, DEFAULTS.L, DEFAULTS.k, DEFAULTS.x0);
            highlightSelectedPoint(selectedDay);

            // Auto-render any inline LaTeX strings that weren't targeted explicitly
            if (typeof renderMathInElement !== 'undefined') {
                renderMathInElement(document.body, {
                    delimiters: [
                        {left: '$$', right: '$$', display: true},
                        {left: '$', right: '$', display: false}
                    ]
                });
            }
        };
